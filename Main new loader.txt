repeat task.wait() until game:IsLoaded()

local RunService = game:GetService("RunService")

getgenv().SL_HUB_LOADED = true

local correctKey = "SLONTOP"

end)

RunService.RenderStepped:Connect(function()
	Title.TextColor3 = Color3.fromHSV(hue,1,1)
end)

local KeyBox = Instance.new("TextBox", KeyFrame)
KeyBox.Size = UDim2.new(0.8,0,0,40)

function OpenMainMenu()
	local MenuGui = Instance.new("ScreenGui", game.CoreGui)
	MenuGui.Name = "SL_MAIN_MENU"

	local Frame = Instance.new("Frame", MenuGui)
	Frame.Size = UDim2.new(0,400,0,320)
	Frame.Position = UDim2.new(0.5,-200,0.5,-160)
	Frame.BackgroundColor3 = Color3.fromRGB(20,20,20)
	Frame.Active = true
	Frame.Draggable = true
	Instance.new("UICorner", Frame)

	local Stroke2 = Instance.new("UIStroke", Frame)
	Stroke2.Thickness = 3

	local hue2 = 0
	RunService.RenderStepped:Connect(function()
		hue2 = (hue2 + 0.003) % 1
		Stroke2.Color = Color3.fromHSV(hue2,1,1)
	end)

	local Title2 = Instance.new("TextLabel", Frame)
	Title2.Size = UDim2.new(1,0,0,50)
	Title2.BackgroundTransparency = 1
	Title2.Text = "SL HUB"
	Title2.Font = Enum.Font.GothamBlack
	Title2.TextSize = 26
	RunService.RenderStepped:Connect(function()
		Title2.TextColor3 = Color3.fromHSV(hue2,1,1)
	end)

	local Holder = Instance.new("Frame", Frame)
	Holder.Size = UDim2.new(1,0,1,-60)
	Holder.Position = UDim2.new(0,0,0,60)
	Holder.BackgroundTransparency = 1

	local UIList = Instance.new("UIListLayout", Holder)
	UIList.Padding = UDim.new(0,12)
	UIList.HorizontalAlignment = Enum.HorizontalAlignment.Center
	UIList.VerticalAlignment = Enum.VerticalAlignment.Center

	local function GameButton(name, url)
		local btn = Instance.new("TextButton", Holder)
		btn.Size = UDim2.new(0.85,0,0,50)
		btn.Text = name
		btn.Font = Enum.Font.GothamBold
		btn.TextSize = 16
		btn.BackgroundColor3 = Color3.fromRGB(35,35,35)
		btn.TextColor3 = Color3.new(1,1,1)
		Instance.new("UICorner", btn)

		RunService.RenderStepped:Connect(function()
			btn.TextColor3 = Color3.fromHSV(hue2,1,1)
		end)

		btn.MouseButton1Click:Connect(function()
			MenuGui:Destroy()
			loadstring(game:HttpGet(url))()
		end)
	end

	GameButton("Legends Of Speed", "https://raw.githubusercontent.com/XMASFIRE/SL-HUB-D/refs/heads/main/SL%20HUB%20LOS")
	GameButton("Muscle Legends", "https://raw.githubusercontent.com/XMASFIRE/SL-HUB-D/refs/heads/main/SL%20HUB%20ML%20(1).txt")
	GameButton("Blox Fruits", "https://raw.githubusercontent.com/XMASFIRE/SL-HUB-D/refs/heads/main/BloxFruits")
	GameButton("Escape Tsunami", "https://raw.githubusercontent.com/XMASFIRE/SL-HUB-D/refs/heads/main/Tusnammi.txt")
end

Confirm.MouseButton1Click:Connect(function()
	if KeyBox.Text == correctKey then
		KeyGui:Destroy()
		OpenMainMenu()
	else
		KeyBox.Text = "Wrong Key!"
		task.wait(1)
		KeyBox.Text = ""
	end
end)

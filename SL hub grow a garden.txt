function CalculatePlantValue(plantInstance)
	local itemString = plantInstance:FindFirstChild("Item_String") or plantInstance.Name;
	if not itemString then
		return 0;
	end;
	local variant = plantInstance:FindFirstChild("Variant");
	if not variant then
		return 0;
	end;
	local weight = plantInstance:FindFirstChild("Weight");
	if not weight then
		return 0;
	end;
	local itemData = ItemModule.Return_Data(itemString.Value or plantInstance.Name);
	if not itemData or #itemData < 3 then
		warn("CalculatePlantValue | ItemData is invalid");
		return 0;
	end;
	local baseValue = itemData[3];
	local multiplier = itemData[2];
	local variantMultiplier = ItemModule.Return_Multiplier(variant.Value);
	local mutationMultiplier = MutationHandler:CalcValueMulti(plantInstance);
	local finalValue = baseValue * mutationMultiplier * variantMultiplier;
	local weightValue = weight.Value;
	local clampedWeight = math.clamp(weightValue / multiplier, 0.95, 100000000);
	local resultValue = finalValue * clampedWeight ^ 2;
	return math.round(resultValue);
end;

local function GetLowestPosition(model)
	if not model or (not model:IsA("Model")) then
		return nil;
	end;
	local lowestPos = nil;
	for _, part in ipairs(model:GetDescendants()) do
		if part:IsA("BasePart") then
			local partBottomPos = part.Position - Vector3.new(0, part.Size.Y / 2, 0);
			if lowestPos == nil or partBottomPos.Y < lowestPos.Y then
				lowestPos = partBottomPos;
			end;
		end;
	end;
	return lowestPos;
end;

function generateRarityColor(rarity)
	if rarity == "Common" then
		return "255, 255, 255";
	elseif rarity == "Uncommon" then
		return "0, 255, 0";
	elseif rarity == "Rare" then
		return "0, 0, 255";
	elseif rarity == "Legendary" then
		return "255, 255, 0";
	elseif rarity == "Mythical" then
		return "170, 85, 255";
	elseif rarity == "Divine" then
		return "255, 85, 0";
	elseif rarity == "Prismatic" then
		return "255, 0, 255";
	else
		return "255, 255, 255";
	end;
end;
function sortItemsByRarity(items)
	local rarityOrder = {
		"Common",
		"Uncommon",
		"Rare",
		"Legendary",
		"Mythical",
		"Divine",
		"Prismatic"
	};
	local function getRarityIndex(rarity)
		for i, v in ipairs(rarityOrder) do
			if v == rarity then
				return i;
			end;
		end;
		return (#rarityOrder) + 1;
	end;
	table.sort(items, function(a, b)
		return getRarityIndex(a.Main_Frame.Rarity_Text.Text) < getRarityIndex(b.Main_Frame.Rarity_Text.Text);
	end);
	return items;
end;

		if v.Important.Data.Owner.Value == game.Players.LocalPlayer.Name then
			return v;
		end;
	end;
end;
function getAllFarm()
	local res = {};
	for i, v in pairs((game:GetService("Workspace")).Farm:GetChildren()) do
		table.insert(res, v);
	end;
	return res;
end;
function getMiddlePosFarm()
	local totalPos = Vector3.new(0, 0, 0);
	local count = 0;
	for i, v in pairs((getOwrnerFarm()).Important.Plant_Locations:GetChildren()) do
		if v:IsA("BasePart") then
			totalPos += v.Position;
			count += 1;
		end;
	end;
	local middlePos;
	if count > 0 then
		middlePos = totalPos / count;
	end;
	return middlePos;
end;
function refactorFruits(plants)
	local result = {};
	for i, v in pairs(plants) do
		if isMultiHarvest(v) then
			local Fruits = v:FindFirstChild("Fruits");
			for i, v in pairs(Fruits:GetChildren()) do
				table.insert(result, v);
			end;
		else
			table.insert(result, v);
		end;
	end;
	return result;
end;
function findNotification(name)
	for i, v in pairs((game:GetService("Players")).LocalPlayer.PlayerGui.Top_Notification.Frame:GetChildren()) do
		if v:FindFirstChild("TextLabel") and string.find(v.TextLabel.Text, name) then
			return v;
		end;
	end;
end;
function destroyNotification(name)
	(findNotification(name)):Destroy();
end;
function CollectFruit(fruit)
	if InventoryModule.IsMaxInventory() then
		return;
	end;
	((RemoteEvents:WaitForChild("Crops")):WaitForChild("Collect")):FireServer({
		fruit
	});
end;
function getAllPlantsGarden()
	local plants = {};
	for i, v in pairs((getOwrnerFarm()).Important.Plants_Physical:GetChildren()) do
		table.insert(plants, v);
	end;
	return plants;
end;
local plantList = {};
task.spawn(function()
	for i, v in pairs(getAllPlantsGarden()) do
		table.insert(plantList, v.Name);
	end;
end);
function getAllPlantsServer()
	local plants = {};
	for _, farm in pairs(getAllFarm()) do
		for i, v in pairs(farm.Important.Plants_Physical:GetChildren()) do
			table.insert(plants, v);
		end;
	end;
	return plants;
end;
function getAllFruitsServer()
	return refactorFruits(getAllPlantsServer());
end;
function getAllFruitsGarden()
	return refactorFruits(getAllPlantsGarden());
end;
function findCollectPrompt(obj)
	return obj:FindFirstChild("ProximityPrompt", true);
end;
function collectFruits(prompt)
	if not prompt then
		return;
	end;
	fireproximityprompt(prompt);
end;

function getHeldPlant()
	local char = (game:GetService("Players")).LocalPlayer.Character;
	for i, v in pairs(allPlants) do
		for _, held in pairs(char:GetChildren()) do
			if string.find(held.Name, v) then
				return v;
			end;
		end;
	end;
	return nil;
end;
spawn(function()
	while task.wait() do
		pcall(function()
			if _G.AutoPlant then
				if _G.SelectedPlantMethod == "Character" then
					if getHeldPlant() then
						Plant((game:GetService("Players")).LocalPlayer.Character.HumanoidRootPart.Position, getHeldPlant());
						wait(0.5);
					end;
				elseif _G.SelectedPlantMethod == "Random" then
					for i, v in pairs((getOwrnerFarm()).Important.Plant_Locations:GetChildren()) do
						if getHeldPlant() then
							Plant((getRandomPosPlant(v)).Position, getHeldPlant());
							wait(0.5);
						end;
					end;
				end;
			end;
		end);
	end;
end);

function checkAvailable(table, name)
	for _, v in pairs(table) do
		if v.Name and v.Name == name or v == name then
			return true;
		end;
	end;
	return false;
end;
spawn(function()
	while wait() do
		pcall(function()
			if _G.AutoBuySeed then
				local selectedSeed = _G.SelectedSeed;
				local seedStock = getSeedStock();
				if #seedStock > 0 then
					for i, v in pairs(seedStock) do
						if checkAvailable(selectedSeed, v.Name) then
							buySeedStock(v.Name);
						end;
					end;
				end;
			end;
		end);
	end;
end);

function refreshSeedStock()
	local result = "";
	local seedStock = getSeedStock();
	local sortedByRarity = sortItemsByRarity(seedStock);
	if #sortedByRarity > 0 then
		for i, v in pairs(sortedByRarity) do
			local rarity = setTextColor(v.Main_Frame.Rarity_Text.Text, generateRarityColor(v.Main_Frame.Rarity_Text.Text));
			local getStockCount = string.match(v.Main_Frame.Stock_Text.Text, "X(%d+)");
			local stock = setTextColor("x" .. getStockCount, tostring("255, 255, 255"));
			result = result .. stock .. " " .. v.Name .. " - " .. rarity;
			if i < (#sortedByRarity) then
				result = result .. "\n";
			end;
		end;
	else
		result = "Out of Stock";
	end;
	return result;
end;

function formatMoney(number)
	local formatted = tostring(number);
	while true do
		formatted, k = string.gsub(formatted, "^(-?%d+)(%d%d%d)", "%1,%2");
		if k == 0 then
			break;
		end;
	end;
	return formatted;
end;
